### 找素数，利用 else

```
k = 0
for num in range(2, 101):  # 生成从2到100的序列
    for i in range(2, num):
        if num % i == 0:
            break
    else:
        print('生成的素数有:{}'.format(num))
        k += 1

print('\n100以内的素数共有{}个'.format(k))
```

### 自由输入求素数个数

```
is_notprime = [] #定义空列表，存放非素数
is_prime = []    #定义空列表，存放素数

num = input('Enter some numbers:').split()  # 由键盘输入一些数字，并以空格分隔

num_list = [int(j) for j in num]      # 用列表推导式并加以强制类型转换把输入的字符数字转换成 int 类型数据

for s in num_list:
    for i in range(2, s):
        if s % i == 0:
            is_notprime.append(s)    # 将非素数添加到非素数空列表
            break
    else:
        is_prime.append(s)   #将是素数的数字添加到素数列表

# 输出非素数及素数
print(f'非素数的数字有:{is_notprime}')
print(f'素数的数字有:{is_prime}')

```

### 寻找素数

```
# coding=utf-8

"""
寻找素数
"""

import math

def is_prime(n):
    """
    判断一个数是否是素数
    """
    if n <= 1:
        return False
    for i in range(2, int(math.sqrt(n) + 1)):
        if n % i == 0:
            return False
    return True

if __name__ == "__main__":
    primes = [i for i in range(2,100) if is_prime(i)]    #从 2 开始，因为 1 显然不是质数
    print (primes)
```

### 一元二次方程求解

```
# coding=utf-8

"""
solving a quadratic equation
"""

from __future__ import division
import math

def quadratic_equation(a,b,c):
    delta = b*b - 4*a*c
    if delta<0:
        return False
    elif delta==0:
        return -(b/(2*a))
    else:
        sqrt_delta = math.sqrt(delta)
        x1 = (-b + sqrt_delta)/(2*a)
        x2 = (-b - sqrt_delta)/(2*a)
        return x1, x2

if __name__ == "__main__":
    print ("a quadratic equation: x^2 + 2x + 1 = 0")
    coefficients = []
    a = input('Enter some numbers:').split( )
    for i in a:
        coefficients.append(int(i))
    roots = quadratic_equation(*coefficients)
    if roots:
        print ("the result is:",roots)
    else:
        print ("this equation has no solution.")
```
